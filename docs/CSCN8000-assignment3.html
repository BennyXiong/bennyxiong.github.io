features = ['Age', 'RestingBP', 'Cholesterol', 'MaxHR', 'Oldpeak']

# Loop through all numeric features to get the indices of all outliers
df_clean = df.copy()
all_outlier_indices = set()
threshold = 3

for feature in features:
    if feature not in df_clean.columns:
        continue

    z_scores = zscore(df[feature].dropna())
    abs_z_scores = np.abs(z_scores)
    mask = abs_z_scores >= threshold
    outlier_idx = df[feature].dropna().index[mask]

    all_outlier_indices.update(outlier_idx)

# Drop all rows with any outlier in any method/feature
df_clean = df_clean.drop(index=all_outlier_indices)

# Encode specified categorical features using LabelEncoder.
def Label_encode_categorical_features(df, features):
    df_encoded = df.copy()

    for feature in features:
        if feature in df_encoded.columns:
            le = LabelEncoder()
            df_encoded[feature] = le.fit_transform(df_encoded[feature].astype(str))

    return df_encoded

# label encoding features 'ChestPainType', 'RestingECG', 'ST_Slope'

In[62]:
# Function to train and evaluate models
def evaluate_models(X_train, X_test, y_train, y_test):
    results = {}
    for name, model in models.items():
        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)
        acc = accuracy_score(y_test, y_pred)
        results[name] = acc
    return results
