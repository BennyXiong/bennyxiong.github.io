# A class takes a DataFrame and a list of features as parameters, and creates box plots for each feature to help visualize outliers.
class BoxPlotter:
    def __init__(self, df, features):
        """
        df: pandas DataFrame
        features: list of column names (numerical) to plot
        """
        self.df = df
        self.features = features

    def plot(self, cols=3, figsize_scale=(5, 4)):
        """
        cols: number of columns in subplot grid
        figsize_scale: tuple controlling figure size scaling (width, height)
        """
        n = len(self.features)
        rows = math.ceil(n / cols)

        plt.figure(figsize=(cols * figsize_scale[0], rows * figsize_scale[1]))

        for i, feature in enumerate(self.features, 1):
            if feature not in self.df.columns:
                continue

            plt.subplot(rows, cols, i)
            sns.boxplot(y=self.df[feature])
            plt.title(feature)

        plt.tight_layout()
        plt.show()

# A class that accepts a list of features and removes outliers from a DataFrame using mean, median, or Z-score method
class OutlierRemover:
    def __init__(self, methods=['mean', 'median', 'zscore']):
        self.methods = methods

    def remove_outliers(self, df, features, z_thresh=3):
        """
        df: pandas DataFrame
        features: list of numeric feature names
        z_thresh: threshold for z-score (only used for zscore method)
        returns: DataFrame with outliers removed
        """
        df_clean = df.copy()
        all_outlier_indices = set()

        for method in self.methods:
            for feature in features:
                if feature not in df_clean.columns:
                    continue

                if method == 'mean':
                    outlier_idx = self._get_outliers_by_mean(df_clean, feature)
                elif method == 'median':
                    outlier_idx = self._get_outliers_by_median(df_clean, feature)
                elif method == 'zscore':
                    outlier_idx = self._get_outliers_by_zscore(df_clean, feature, z_thresh)

                all_outlier_indices.update(outlier_idx)

        # Drop all rows with any outlier in any method/feature
        df_clean = df_clean.drop(index=all_outlier_indices)
        return df_clean

    def _get_outliers_by_mean(self, df, feature):
        mean = df[feature].mean()
        std = df[feature].std()
        lower = mean - 3 * std
        upper = mean + 3 * std
        return df[(df[feature] < lower) | (df[feature] > upper)].index

    def _get_outliers_by_median(self, df, feature):
        median = df[feature].median()
        mad = np.median(np.abs(df[feature] - median))
        lower = median - 3 * mad
        upper = median + 3 * mad
        return df[(df[feature] < lower) | (df[feature] > upper)].index

    def _get_outliers_by_zscore(self, df, feature, threshold):
        z_scores = zscore(df[feature].dropna())
        abs_z_scores = np.abs(z_scores)
        mask = abs_z_scores >= threshold
        return df[feature].dropna().index[mask]

Label encoding and one hot encoding¶

  • Categorical features with text data ('Sex', 'ChestPainType', 'RestingECG', 'ExerciseAngina', 'ST_Slope') are good candidates for label encoding and one hot encoding.
  • Label encoding is better for ordinal data with many unique values, e.g. 'ChestPainType', 'RestingECG', and 'ST_Slope'.
  • One hot encoding is better for nominal data and creates a new binary column for each category. Which makes it ideal for 'Sex' and 'ExerciseAngina'.

# Encode specified categorical features using LabelEncoder.
def Label_encode_categorical_features(df, features):
    df_encoded = df.copy()

    for feature in features:
        if feature in df_encoded.columns:
            le = LabelEncoder()
            df_encoded[feature] = le.fit_transform(df_encoded[feature].astype(str))

    return df_encoded

  • Support vector machine (SVM) classifier, both standalone and bagging, has better performance than Decision Tree classifier (standalone and bagging).
  • Bagging with SVM has little effect on the performance, but it improve the model's consistency. This is because bagging helps unstable models, but SVMs are generally stable. The model is not very sensitive to small data changes, so each bagged model is similar.
  • Bagging with Decision Tree improves performance and still keeps the model's consistency. This is because Decision Trees are high-variance models, meaning small changes in data can lead to very different trees. Bagging trains multiple trees on random subsets of the data and averages their predictions, smoothing out those wild changes.
